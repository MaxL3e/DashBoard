                      const context = this, args = arguments;

                      timeout = setTimeout(() => func.apply(context, args), wait);

              // 迷你趋势图
              function initMiniCharts() {
                  // 简化版迷你图表，使用canvas绘制简单趋势线
                  const drawMiniChart = (id, data, color) => {
                      const canvas = document.getElementById(id);
                      if (!canvas) return;

                      const ctx = canvas.getContext('2d');
                      const width = canvas.width;
                      const height = canvas.height;

                      // 清除画布
                      ctx.clearRect(0, 0, width, height);

                      // 设置数据
                      const points = data;
                      const pointCount = points.length;
                      const segmentWidth = width / (pointCount - 1);
                      const maxValue = Math.max(...points);
                      const minValue = Math.min(...points);
                      const valueRange = maxValue - minValue || 1; // 避免除以零

                      // 绘制区域填充
                      ctx.beginPath();
                      ctx.moveTo(0, height);

                      points.forEach((value, i) => {
                          const x = i * segmentWidth;
                          const y = height - ((value - minValue) / valueRange) * height * 0.8; // 留一些顶部空间

                          if (i === 0) {
                              ctx.moveTo(x, y);
                          } else {
                              ctx.lineTo(x, y);
                          }
                      });

                      ctx.lineTo((pointCount - 1) * segmentWidth, height);
                      ctx.closePath();
                      ctx.fillStyle = color + '20'; // 添加透明度
                      ctx.fill();

                      // 绘制线条
                      ctx.beginPath();

                      points.forEach((value, i) => {
                          const x = i * segmentWidth;
                          const y = height - ((value - minValue) / valueRange) * height * 0.8;

                          if (i === 0) {
                              ctx.moveTo(x, y);
                          } else {
                              ctx.lineTo(x, y);
                          }
                      });

                      ctx.strokeStyle = color;
                      ctx.lineWidth = 2;
                      ctx.stroke();

                      // 绘制最后一个点
                      const lastValue = points[points.length - 1];
                      const lastX = (pointCount - 1) * segmentWidth;
                      const lastY = height - ((lastValue - minValue) / valueRange) * height * 0.8;

                      ctx.beginPath();
                      ctx.arc(lastX, lastY, 3, 0, Math.PI * 2);
                      ctx.fillStyle = color;
                      ctx.fill();
                  };

                  // 为每个迷你图表绘制数据
                  drawMiniChart('projectChart', [12, 15, 14, 18, 20, 22, 24], '#165DFF');
                  drawMiniChart('positionChart', [76, 72, 80, 85, 90, 88, 98], '#36CFC9');
                  drawMiniChart('resumeChart', [1856, 1920, 2050, 2180, 2250, 2380, 2456], '#52C41A');
                  drawMiniChart('hiredChart', [175, 182, 190, 185, 192, 180, 177], '#FAAD14');
              }
